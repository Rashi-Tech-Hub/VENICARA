  <script>
    const smallImg = document.getElementById("smallImg");
const lens = document.getElementById("lens");
const zoomBox = document.getElementById("zoomBox");

smallImg.addEventListener("mouseenter", () => lens.style.display = "block");
smallImg.addEventListener("mouseleave", () => lens.style.display = "none");

smallImg.addEventListener("mousemove", function(e) {
  const rect = smallImg.getBoundingClientRect();
  let x = e.clientX - rect.left;
  let y = e.clientY - rect.top;
  // Move lens
  lens.style.left = x - lens.offsetWidth / 2 + "px";
  lens.style.top = y - lens.offsetHeight / 2 + "px";
  
  console.log(x,y)
  // Show zoomed image
  zoomBox.style.backgroundImage = `url('Yves Saint Laurent Black Opium (3).webp')`;
  zoomBox.style.backgroundSize = `${smallImg.width * 3}px ${smallImg.height * 3}px`;
  zoomBox.style.backgroundPosition = 
    `-${x * 3 - zoomBox.offsetWidth/2}px -${y * 3 - zoomBox.offsetHeight/2}px`;
});

  </script>
